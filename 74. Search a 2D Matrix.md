        if not matrix:
            return False
        
        m, n = len(matrix), len(matrix[0])
        
        
        if m <= 1:
            return self.binarySearch(matrix[0] ,0,n-1,target)
        
        mid = matrix[int(m/2)]
        if mid[int(n/2)] == target:
            return True
        elif mid[int(n/2)] > target:
            if mid[0] <= target:
                return self.binarySearch(mid,0,int(n/2),target)
            else:
                return self.searchMatrix(matrix[:int(m/2)],target)
        elif mid[int(n/2)] < target:
            if mid[n-1] >= target:
                return self.binarySearch(mid,int(n/2),n-1,target)

            return False

            return True
        elif matrix[mid] < target:
            return self.binarySearch(matrix,mid+1,end,target)
        else:
            return self.binarySearch(matrix,start,mid-1,target)

class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        
        if not matrix:
            return False
        
        n = len(matrix[0])
        lo, hi = 0, len(matrix) * n
        while lo < hi:
            mid = int((lo + hi) / 2)
            x = matrix[int(mid/n)][mid%n]
            if x < target:
                lo = mid + 1
            elif x > target:
                hi = mid
            else:
                return True
        return False

class Solution:

    # @param target, an integer

    def searchMatrix(self, matrix, target):
        if not matrix or target is None:
            return False

        rows, cols = len(matrix), len(matrix[0])
        low, high = 0, rows * cols - 1
        
        while low <= high:
            mid = (low + high) / 2
            num = matrix[mid / cols][mid % cols]

            if num == target:
                return True
            elif num < target:
                low = mid + 1
            else:
                high = mid - 1
        
        return False
